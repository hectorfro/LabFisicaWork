\begin{frame}
\frametitle{Introducción}
\begin{itemize}
    \item En este capítulo se explorarán los tipos de errores en las mediciones experimentales.
    \item Se discutirán estrategias para minimizar y cuantificar estos errores.
    \item Además, se introducirán métodos para calcular incertidumbres.
    \item Ejemplo: Medición de longitud con diferentes instrumentos.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Exactitud, Precisión y Sensibilidad}
\begin{itemize}
    \item \textbf{Exactitud:} Concordancia entre el valor “verdadero” y el medido.
    \item \textbf{Precisión:} Concordancia entre medidas realizadas en condiciones iguales.
    \item \textbf{Sensibilidad:} Valor mínimo medible por un aparato.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Tipos de Errores en las Mediciones}
\begin{itemize}
    \item \textbf{Errores sistemáticos:} Afectan resultados en una misma dirección. Causas: defectos en instrumentos, errores personales, condiciones ambientales, método empleado.
    \item \textbf{Errores casuales o accidentales:} Producto de fuentes incontrolables, afectan aleatoriamente el valor medido.
    \item \textbf{Errores de precisión:} Determinados por la resolución del aparato de medida.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{La Escritura de los Resultados de una Medición}
\begin{equation}
x = x_0 \pm \Delta x \quad \text{[unidades]}
\end{equation}
\begin{itemize}
    \item $x_0$: valor de la medida.
    \item $\Delta x$: incertidumbre o error de la medida.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Error Absoluto y Error Relativo}
\begin{itemize}
    \item \textbf{Error Absoluto:} $\Delta x = |x - x_0|$
    \item \textbf{Error Relativo:} $\varepsilon = \frac{\Delta x}{x_0}$
    \item \textbf{Error Porcentual:} $\varepsilon \times 100\%$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Mediciones Directas e Indirectas}
\begin{itemize}
    \item \textbf{Mediciones Directas:} Medidas obtenidas directamente del aparato.
    \item \textbf{Mediciones Indirectas:} Determinadas a partir de otras magnitudes medidas directamente.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Cálculo de los Errores en una Serie de Medidas}
\begin{itemize}
    \item \textbf{Valor Medio Aritmético:} $\bar{x} = \frac{1}{n} \sum_{i=1}^n x_i$
    \item \textbf{Error Medio Absoluto:} $\overline{\Delta x} = \frac{1}{n} \sum_{i=1}^n \Delta x_i$
    \item \textbf{Error Relativo:} $\varepsilon_x = \frac{\overline{\Delta x}}{\bar{x}}$
    \item \textbf{Error Porcentual:} $\varepsilon_{\%} = \varepsilon_x \times 100$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Desviación Estándar y Error Probable}
\begin{itemize}
    \item \textbf{Desviación Estándar:} $\sigma = \sqrt{\frac{1}{n} \sum_{i=1}^n (x_i - \bar{x})^2}$
    \item \textbf{Error Probable:} $\varepsilon_p = 0.674 \sigma$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Ejemplo: Medición del Diámetro de un Tubo}
\begin{itemize}
    \item Diámetro medido 5 veces con un vernier:
    \item $d = [1.780, 1.780, 1.780, 1.790, 1.790] \, \text{cm}$
    \item Diámetro promedio: $\bar{d} = 1.784 \, \text{cm}$
    \item Dispersión: $D = 1.79 - 1.78 = 0.01$
    \item Dispersión porcentual: $D_{\%} = \frac{0.01}{1.784} \times 100 = 0.6 \%$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Propagación de Errores}
\begin{itemize}
    \item \textbf{Suma y Resta:} $\Delta M = \Delta X + \Delta Y$
    \item \textbf{Multiplicación y División:} $\varepsilon_r = \frac{\Delta X}{X} + \frac{\Delta Y}{Y}$
    \item \textbf{Función Potencial:} $\varepsilon_r = n \frac{\Delta X}{X}$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Método del Binomio y Método de las Derivadas Parciales}
\begin{itemize}
    \item \textbf{Método del Binomio:} $\varepsilon_r = a \frac{\Delta X}{X} + b \frac{\Delta Y}{Y} + c \frac{\Delta Z}{Z} + d \frac{\Delta T}{T}$
    \item \textbf{Método de las Derivadas Parciales:} $\Delta M = \left|\frac{\partial M}{\partial X}\right|\Delta X + \left|\frac{\partial M}{\partial Y}\right|\Delta Y + \left|\frac{\partial M}{\partial Z}\right|\Delta Z + \left|\frac{\partial M}{\partial T}\right|\Delta T$
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Ejercicio 1: Cálculo del Error en el Módulo de Rigidez (G)}
\begin{itemize}
    \item \textbf{Datos:}
    \begin{itemize}
        \item $L = (104.2 \pm 0.1) \, \text{cm}$
        \item $R = (0.03005 \pm 0.00003) \, \text{cm}$
        \item $k = (9.92 \pm 0.02) \times 10^3 \, \text{dyn cm}$
    \end{itemize}
    \item \textbf{Fórmula:}
    \begin{equation}
    G = \frac{2 L k}{\pi R^4}
    \end{equation}
    \item \textbf{Errores Relativos:}
    \begin{equation}
    \frac{\Delta G}{G} = \frac{\Delta L}{L} + \frac{\Delta k}{k} + 4 \frac{\Delta R}{R}
    \end{equation}
    \begin{equation}
    \frac{\Delta G}{G} = \frac{0.1}{104.2} + \frac{0.02}{9.92} + 4 \frac{0.00003}{0.03005}
    \end{equation}
    \begin{equation}
    \frac{\Delta G}{G} \approx 0.001 + 0.002 + 0.004
    \end{equation}
    \item \textbf{Valor de G:}
    \begin{equation}
    G = \frac{2 \times 104.2 \times 9.92 \times 10^3}{\pi \times (0.03005)^4} \approx 1.15 \times 10^9 \, \text{dyn/cm}^2
    \end{equation}
    \item \textbf{Error Absoluto:}
    \begin{equation}
    \Delta G \approx 0.007 \times 1.15 \times 10^9 \approx 8.05 \times 10^6 \, \text{dyn/cm}^2
    \end{equation}
    \item \textbf{Resultado Final:}
    \begin{equation}
    G = (1.15 \pm 0.01) \times 10^9 \, \text{dyn/cm}^2
    \end{equation}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Ejercicio 2: Perímetro y Área de una Plancha Metálica}
\begin{itemize}
    \item \textbf{Datos:}
    \begin{itemize}
        \item Largo: $L = (53.154 \pm 0.3) \, \text{cm}$
        \item Ancho: $A = 12.5 \, \text{cm}$ con $4\%$ de error
    \end{itemize}
    \item \textbf{Perímetro:}
    \begin{equation}
    P = 2(L + A) = 2(53.154 + 12.5) = 131.308 \, \text{cm}
    \end{equation}
    \item \textbf{Error en el Perímetro:}
    \begin{equation}
    \Delta P = 2 (\Delta L + \Delta A) = 2 (0.3 + 0.5) = 1.6 \, \text{cm}
    \end{equation}
    \begin{equation}
    P = 131.3 \pm 1.6 \, \text{cm}
    \end{equation}
    \item \textbf{Área:}
    \begin{equation}
    A = L \times A = 53.154 \times 12.5 = 664.425 \, \text{cm}^2
    \end{equation}
    \item \textbf{Error en el Área:}
    \begin{equation}
    \frac{\Delta A}{A} = \frac{\Delta L}{L} + \frac{\Delta A}{A}
    \end{equation}
    \begin{equation}
    \frac{\Delta A}{664.425} = \frac{0.3}{53.154} + 0.04
    \end{equation}
    \begin{equation}
    \Delta A \approx 26.5 \, \text{cm}^2
    \end{equation}
    \item \textbf{Resultado Final:}
    \begin{equation}
    A = 664 \pm 27 \, \text{cm}^2
    \end{equation}
\end{itemize}
\end{frame}
